using System;
using System.IO;
using System.Text;

class QuakeMdlWriter
{
    // Minimal Quake palette (256*3 bytes). Replace with full palette for real skins.
    static byte[] palette = new byte[256 * 3];

    static void Main()
    {
        // Minimal model: 1 skin (8x8), 1 texcoord, 1 triangle, 1 frame, 1 vertex
        string outPath = "minimal_quad.mdl";
        using (var fs = new FileStream(outPath, FileMode.Create, FileAccess.Write))
        using (var bw = new BinaryWriter(fs))
        {
            // We'll build content sections in memory to compute offsets
            var skins = BuildSkins();
            var texcoords = BuildTexCoords();
            var triangles = BuildTriangles();
            var frames = BuildFrames();

            int ident = 0x54445349; // 'IDST' little-endian
            int version = 6;
            int skinWidth = 8;
            int skinHeight = 8;
            int numSkins = 1;
            int numVerts = 1;
            int numTris = 1;
            int numFrames = 1;
            int synctype = 0;
            int flags = 0;
            float size = 1.0f;

            // Header is 17*4 + (3*4 for bounding box?) — use known layout:
            // typedef struct {
            //   int ident; int version;
            //   char name[64];
            //   int length;
            //   float eyeposition[3];
            //   float min[3], max[3], bbmin[3], bbmax[3];
            //   int num_skins, skinwidth, skinheight, num_verts, num_tris, num_frames, synctype, flags, size;
            //   int ofs_skins, ofs_st, ofs_tris, ofs_frames, ofs_glcmds, ofs_end;
            // } mdl_t;
            // We'll write name and placeholders then fill offsets.

            long headerPos = bw.BaseStream.Position;
            bw.Write(ident);
            bw.Write(version);

            // name[64]
            var nameBytes = new byte[64];
            Encoding.ASCII.GetBytes("minimal").CopyTo(nameBytes, 0);
            bw.Write(nameBytes);

            // placeholder length (int)
            bw.Write(0);

            // eyeposition[3], min[3], max[3], bbmin[3], bbmax[3] -- write zeros
            for (int i = 0; i < 3 * 5; i++) bw.Write(0.0f);

            bw.Write(numSkins);
            bw.Write(skinWidth);
            bw.Write(skinHeight);
            bw.Write(numVerts);
            bw.Write(numTris);
            bw.Write(numFrames);
            bw.Write(synctype);
            bw.Write(flags);
            bw.Write(size);

            // placeholders for offsets
            int ofs_skins_pos = (int)bw.BaseStream.Position;
            bw.Write(0); // ofs_skins
            bw.Write(0); // ofs_st
            bw.Write(0); // ofs_tris
            bw.Write(0); // ofs_frames
            bw.Write(0); // ofs_glcmds
            bw.Write(0); // ofs_end

            // Now compute and write sections, capturing offsets
            int ofs_skins = (int)bw.BaseStream.Position;
            bw.Write(skins);
            int ofs_st = (int)bw.BaseStream.Position;
            bw.Write(texcoords);
            int ofs_tris = (int)bw.BaseStream.Position;
            bw.Write(triangles);
            int ofs_frames = (int)bw.BaseStream.Position;
            bw.Write(frames);
            int ofs_glcmds = (int)bw.BaseStream.Position;
            // write empty glcmds (0 terminated)
            bw.Write(0);
            int ofs_end = (int)bw.BaseStream.Position;

            // file length
            int length = ofs_end;

            // Seek back to fill length and offsets
            bw.BaseStream.Seek(headerPos + 4 + 64, SeekOrigin.Begin); // after ident+version+name
            bw.Write(length);

            bw.BaseStream.Seek(ofs_skins_pos, SeekOrigin.Begin);
            bw.Write(ofs_skins);
            bw.Write(ofs_st);
            bw.Write(ofs_tris);
            bw.Write(ofs_frames);
            bw.Write(ofs_glcmds);
            bw.Write(ofs_end);

            bw.Flush();
            Console.WriteLine("Wrote " + outPath);
        }
    }

    static byte[] BuildSkins()
    {
        using (var ms = new MemoryStream())
        using (var bw = new BinaryWriter(ms))
        {
            // skin group flag (0 = single skin)
            bw.Write(0);
            // skin data: width*height bytes (8x8)
            int w = 8, h = 8;
            for (int i = 0; i < w * h; i++)
            {
                // fill with palette index 1
                bw.Write((byte)1);
            }
            // write palette (768 bytes)
            bw.Write(palette);
            return ms.ToArray();
        }
    }

    static byte[] BuildTexCoords()
    {
        using (var ms = new MemoryStream())
        using (var bw = new BinaryWriter(ms))
        {
            // st entry: short s, short t, short onseam (or seam?)
            // For Quake MDL, st is two shorts and a seam flag as short.
            short s = 0, t = 0, onseam = 0;
            bw.Write(s);
            bw.Write(t);
            bw.Write(onseam);
            return ms.ToArray();
        }
    }

    static byte[] BuildTriangles()
    {
        using (var ms = new MemoryStream())
        using (var bw = new BinaryWriter(ms))
        {
            // triangle struct: short facesfront; short vertex[3];
            short facesfront = 0;
            bw.Write(facesfront);
            // vertices indices (3). For minimal, repeat same vertex index 0.
            bw.Write((short)0);
            bw.Write((short)0);
            bw.Write((short)0);
            return ms.ToArray();
        }
    }

    static byte[] BuildFrames()
    {
        using (var ms = new MemoryStream())
        using (var bw = new BinaryWriter(ms))
        {
            // frame type (0 = single)
            bw.Write(0);

            // bboxmins (3 bytes) and bboxmaxs (3 bytes) - but stored as bytes in mdl frames
            // In header we reserved floats for bbox; in frames we write bytes for bbox
            bw.Write((sbyte)0); bw.Write((sbyte)0); bw.Write((sbyte)0); // mins
            bw.Write((sbyte)1); bw.Write((sbyte)1); bw.Write((sbyte)1); // maxs

            // frame name (16 bytes)
            var nameB = new byte[16];
            Encoding.ASCII.GetBytes("frame0").CopyTo(nameB, 0);
            bw.Write(nameB);

            // scale (3 floats) and translate (3 floats)
            bw.Write(1.0f); bw.Write(1.0f); bw.Write(1.0f);
            bw.Write(0.0f); bw.Write(0.0f); bw.Write(0.0f);

            // number of verts assumed from header; for each vertex: byte v[3]; byte lightnormalindex;
            // We'll write one vertex at (0,0,0) packed bytes
            bw.Write((byte)0); bw.Write((byte)0); bw.Write((byte)0); // v[3]
            bw.Write((byte)0); // lightnormalindex

            return ms.ToArray();
        }
    }
}









typedef struct {
    int   ident;          // 'IDPO' (0x4F504449)
    int   version;        // 6

    char  name[64];       // model name (null-terminated if shorter)
    int   length;         // file length

    float eyeposition[3];
    float min[3];
    float max[3];
    float bbmin[3];
    float bbmax[3];

    int   num_skins;
    int   skinwidth;
    int   skinheight;
    int   num_verts;
    int   num_tris;
    int   num_frames;
    int   synctype;
    int   flags;
    float size;

    int   ofs_skins;      // offset to skins
    int   ofs_st;         // offset to texture coords
    int   ofs_tris;       // offset to triangles
    int   ofs_frames;     // offset to frames
    int   ofs_glcmds;     // offset to GL command list
    int   ofs_end;        // offset to end of file
} mdl_t;










The Quake **.MDL (version 6)** format (used in Quake) is a compact binary structure for animated 3D models. It’s very structured—basically a header followed by skins, texture coordinates, triangles, and animation frames.

Here’s a clean byte-level breakdown so you can parse it correctly.

---

# 🔹 1. Header (84 bytes total)

All values are **little-endian**.

| Offset | Size | Type     | Name           | Description                  |
| ------ | ---- | -------- | -------------- | ---------------------------- |
| 0      | 4    | int      | ident          | Magic: `"IDPO"` (0x4F504449) |
| 4      | 4    | int      | version        | Must be `6`                  |
| 8      | 12   | float[3] | scale          | Scale factors (x, y, z)      |
| 20     | 12   | float[3] | translate      | Position offset              |
| 32     | 4    | float    | boundingradius | Model bounding radius        |
| 36     | 12   | float[3] | eyeposition    | Eye position                 |
| 48     | 4    | int      | num_skins      | Number of skins              |
| 52     | 4    | int      | skinwidth      | Texture width                |
| 56     | 4    | int      | skinheight     | Texture height               |
| 60     | 4    | int      | num_verts      | Vertices per frame           |
| 64     | 4    | int      | num_tris       | Number of triangles          |
| 68     | 4    | int      | num_frames     | Number of frames             |
| 72     | 4    | int      | synctype       | 0 = sync, 1 = random         |
| 76     | 4    | int      | flags          | Rendering flags              |
| 80     | 4    | float    | size           | Average triangle size        |

---

# 🔹 2. Skins

Immediately after the header.

Each skin starts with:

| Size | Type | Name  |
| ---- | ---- | ----- |
| 4    | int  | group |

### If `group == 0` (single skin):

* Followed by:
  `skinwidth * skinheight` bytes (8-bit indexed texture)

### If `group != 0` (animated skin group):

| Size          | Type    | Description              |
| ------------- | ------- | ------------------------ |
| 4             | int     | num_skins                |
| num_skins * 4 | float[] | time intervals           |
| ...           | byte[]  | pixel data for each skin |

---

# 🔹 3. Texture Coordinates (ST Vertices)

Array of `num_verts` entries:

| Size | Type | Name   |
| ---- | ---- | ------ |
| 4    | int  | onseam |
| 4    | int  | s      |
| 4    | int  | t      |

Total per vertex: **12 bytes**

---

# 🔹 4. Triangles

Array of `num_tris` entries:

| Size | Type   | Name           |
| ---- | ------ | -------------- |
| 4    | int    | facesfront     |
| 12   | int[3] | vertex indices |

Total per triangle: **16 bytes**

---

# 🔹 5. Frames

This is the most complex part.

Each frame begins with:

| Size | Type | Name |
| ---- | ---- | ---- |
| 4    | int  | type |

---

## 🔸 Frame Types

### ➤ Type 0: Single Frame

Structure:

### Frame Header:

| Size | Type     | Name    |
| ---- | -------- | ------- |
| 4    | byte[4]  | bboxmin |
| 4    | byte[4]  | bboxmax |
| 16   | char[16] | name    |

⚠️ `bboxmin/max` are actually:

* 3 bytes: compressed vertex coords
* 1 byte: normal index

---

### Vertex Data (per vertex):

Each vertex is:

| Size | Type    | Name                    |
| ---- | ------- | ----------------------- |
| 3    | byte[3] | v (compressed position) |
| 1    | byte    | lightnormalindex        |

Total per vertex: **4 bytes**

Total frame size:

```
(4 + 4 + 16) + num_verts * 4
```

---

### ➤ Type 1: Group Frame (Animated)

| Size           | Type    | Description               |
| -------------- | ------- | ------------------------- |
| 4              | int     | num_frames                |
| num_frames * 4 | float[] | intervals                 |
| ...            | frames  | sequence of simple frames |

Each sub-frame uses the same structure as Type 0.

---

# 🔹 6. Vertex Decompression Formula

Stored vertex coordinates are bytes, so you reconstruct like:

```
real_x = v[0] * scale[0] + translate[0]
real_y = v[1] * scale[1] + translate[1]
real_z = v[2] * scale[2] + translate[2]
```

---

# 🔹 7. Normals

The `lightnormalindex` is an index into a predefined table of **162 normals** (hardcoded in Quake).

---

# 🔹 8. Key Notes / Gotchas

* Everything is **little-endian**
* No padding between sections
* Skins are **8-bit paletted**
* Texture seam handling depends on `onseam` + `facesfront`
* Frames are **vertex-animated**, not skeletal
* Normals are indexed, not stored as vectors

---

# 🔹 Minimal Parsing Order

1. Read header
2. Read skins
3. Read ST vertices
4. Read triangles
5. Read frames
